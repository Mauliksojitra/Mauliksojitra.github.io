What is a Class?

A class is a blueprint for creating objects — it defines data (fields/properties) and behavior (methods) bundled together. It is the foundation of Object-Oriented Programming in C#.

Car.cs
class Car
{
    // Fields (data)
    public string Brand;
    public int    Speed;

    // Constructor
    public Car(string brand, int speed)
    {
        Brand = brand;
        Speed = speed;
    }

    // Method (behavior)
    public void Drive()
        => Console.WriteLine($"{Brand} is driving at {Speed} km/h");
}

// Creating an object (instance)
Car myCar = new Car("Toyota", 120);
myCar.Drive(); // Toyota is driving at 120 km/h
02

8 Types of Classes

🔷

Regular Class

Standard class. Can be instantiated and inherited freely.

✓ Instantiate✓ Inherit
class Animal
{
    public string Name;
    public void Speak()
        => Console.WriteLine($"{Name} speaks");
}
Animal a = new Animal(); // ✅ OK
🔶

Abstract Class

Cannot instantiate directly. Forces subclasses to implement abstract members.

✗ Instantiate✓ Inherit
abstract class Shape
{
    public abstract double Area();
}
class Circle : Shape
{
    public double Radius;
    public override double Area()
        => Math.PI * Radius * Radius;
}
🔒

Sealed Class

Cannot be inherited. Used to prevent extension for security or performance.

✓ Instantiate✗ Inherit
sealed class BankAccount
{
    public double Balance { get; set; }
}
// class Savings : BankAccount { } ❌
📦

Static Class

Only static members. No instantiation or inheritance. Used for utilities.

✗ Instantiate✗ Inherit
static class MathHelper
{
    public static int Square(int x) => x * x;
    public static int Cube(int x) => x * x * x;
}
MathHelper.Square(4); // → 16
🧩

Partial Class

Splits a class across multiple files. All parts merge at compile time.

✓ Instantiate✓ Inherit
// File1.cs
partial class Employee
{ public string Name; }

// File2.cs
partial class Employee
{
    public void Display()
        => Console.WriteLine(Name);
}
🪆

Nested Class

Defined inside another class. Groups tightly-related logic together.

✓ Instantiate✓ Inherit
class Outer
{
    private int x = 10;
    class Inner
    {
        public void Show()
            => Console.WriteLine("Inner class");
    }
}
🏗️

Generic Class

Works with any data type using type parameters <T>. Type-safe and reusable.

✓ Instantiate✓ Inherit
class Box<T>
{
    public T Value;
    public void Show()
        => Console.WriteLine(Value);
}
Box<int> b = new() { Value = 42 };
📋

Interface

Defines a contract. All members must be implemented by the class.

✗ Instantiate✓ Implement
interface IFlyable
{
    void Fly();
}
class Bird : IFlyable
{
    public void Fly()
        => Console.WriteLine("Flying!");
}

Access Modifiers

public — Accessible from anywhere. No restrictions. Most open access level.
AccessModifiers.cs
class BankAccount
{
    public             string  AccountNumber;  // Anyone can read/write
    private            double  _balance;       // Only this class
    protected          string  OwnerName;      // This class + subclasses
    internal           bool    IsActive;       // Same assembly only
    protected internal int     BranchCode;     // Derived OR same assembly
    private protected  double  InterestRate;   // Derived AND same assembly

    public void Deposit(double amount)
    {
        _balance += amount; // private field used safely inside class
    }

    public double GetBalance() => _balance; // expose via public method
}

Class Modifiers

abstract
sealed
static
partial
abstract — Cannot be instantiated. Must be subclassed. May contain abstract members with no body.
ClassModifiers.cs
// abstract — template class, cannot create object directly
public abstract class Vehicle
{
    public abstract void Start();
    public void Stop() => Console.WriteLine("Stopped");
}

// sealed — prevents all inheritance
public sealed class Car : Vehicle
{
    public override void Start() => Console.WriteLine("Car started");
}

// static — utility class, no instance possible
public static class StringUtils
{
    public static string Reverse(string s)
        => new string(s.ToCharArray().Reverse().ToArray());
}

// partial — class split across files (common in generated code)
public partial class Order { public int Id; }       // File 1
public partial class Order { public void Ship() {} }  // File 2

static — Belongs to the class itself, not an instance. Access via ClassName.Member.
MemberModifiers.cs
abstract class Shape
{
    public  const    string Category = "Geometry";    // compile-time constant
    public  static   int    Count    = 0;             // shared across all instances
    public  readonly Guid   Id;                       // set once in constructor

    public abstract double Area();          // no body — MUST be overridden
    public virtual  string Describe()      // has body — CAN be overridden
        => "I am a shape";
}

class Circle : Shape
{
    public double Radius;
    public override double Area()
        => Math.PI * Radius * Radius;       // override abstract
    public override string Describe()
        => "I am a circle";                // override virtual
}

class SpecialCircle : Circle
{
    public new double Radius;              // hides parent's Radius
    public sealed override string Describe()
        => "I am special";                // blocks further override
}
06

Constructor Types

Default Constructor — Takes no parameters. If you define no constructor at all, C# automatically provides this. Sets all fields to their default values (0, null, false).
Default.cs
class Person
{
    public string Name;
    public int    Age;

    // Default constructor — no parameters
    public Person()
    {
        Name = "Unknown";
        Age  = 0;
    }
}

// Usage
Person p = new Person(); // Name = "Unknown", Age = 0
Console.WriteLine(p.Name); // Unknown
Parameterized Constructor — Accepts one or more arguments. Allows setting field values at the time of object creation. Most commonly used constructor type.
Parameterized.cs
class Person
{
    public string Name;
    public int    Age;
    public string City;

    // 2-param constructor
    public Person(string name, int age)
    {
        Name = name;
        Age  = age;
    }

    // 3-param constructor (overloaded)
    public Person(string name, int age, string city)
    {
        Name = name;
        Age  = age;
        City = city;
    }
}

Person p1 = new Person("Alice", 30);
Person p2 = new Person("Bob", 25, "London");
Copy Constructor — Takes an instance of the same class and copies all its field values into the new object. C# doesn't generate this automatically — you write it manually.
Copy.cs
class Person
{
    public string Name;
    public int    Age;

    public Person(string name, int age)
    {
        Name = name;
        Age  = age;
    }

    // Copy constructor — takes same type as param
    public Person(Person source)
    {
        Name = source.Name;
        Age  = source.Age;
    }
}

Person original = new Person("Alice", 30);
Person copy     = new Person(original); // deep copy

copy.Name = "Bob"; // original.Name still "Alice"
Static Constructor — Runs once before any instance is created or any static member is accessed. Cannot have access modifiers or parameters. Used to initialize static fields.
Static.cs
class Database
{
    public static string  ConnectionString;
    public static int     MaxConnections;
    public string         TableName;

    // Static constructor — no modifier, no params
    static Database()
    {
        ConnectionString = "Server=localhost;DB=mydb";
        MaxConnections   = 100;
        Console.WriteLine("Static constructor called once!");
    }

    // Regular instance constructor
    public Database(string tableName)
    {
        TableName = tableName;
    }
}

// Static ctor fires before first object creation
Database db1 = new Database("Users");
Database db2 = new Database("Orders"); // static ctor NOT called again
Private Constructor — Prevents external code from creating instances. Used in Singleton patterns or utility classes where you want full control over object creation.
Private.cs
class Singleton
{
    private static Singleton _instance;

    // Private constructor — outside code cannot call new
    private Singleton()
    {
        Console.WriteLine("Singleton created!");
    }

    // Controlled access through static factory method
    public static Singleton GetInstance()
    {
        if (_instance == null)
            _instance = new Singleton();
        return _instance;
    }

    public void DoWork() => Console.WriteLine("Working...");
}

// new Singleton() ❌ — won't compile
Singleton s = Singleton.GetInstance(); // ✅
s.DoWork();
Constructor Chaining (this) — One constructor calls another in the same class using : this(...). Avoids code duplication and ensures shared initialization logic runs every time.
Chained.cs
class Product
{
    public string Name;
    public double Price;
    public int    Stock;

    // Master constructor — all params
    public Product(string name, double price, int stock)
    {
        Name  = name;
        Price = price;
        Stock = stock;
    }

    // Chains to master, provides default stock = 0
    public Product(string name, double price)
        : this(name, price, 0) { }

    // Chains to master, provides all defaults
    public Product()
        : this("Unknown", 0.0, 0) { }
}

Product p1 = new Product();                    // all defaults
Product p2 = new Product("Book", 9.99);         // stock = 0
Product p3 = new Product("Pen", 1.50, 100);
Base Constructor (base) — Calls the parent class constructor using : base(...). Ensures the base class is properly initialized before the derived class constructor runs.
Base.cs
class Animal
{
    public string Name;
    public int    Age;

    public Animal(string name, int age)
    {
        Name = name;
        Age  = age;
        Console.WriteLine($"Animal created: {Name}");
    }
}

class Dog : Animal
{
    public string Breed;

    // Calls Animal(name, age) before Dog body runs
    public Dog(string name, int age, string breed)
        : base(name, age)
    {
        Breed = breed;
        Console.WriteLine($"Dog breed: {Breed}");
    }
}

// Output order:
Dog d = new Dog("Rex", 3, "Husky");
// 1. Animal created: Rex
// 2. Dog breed: Husky
Expression-bodied Constructor — A concise constructor using the => arrow syntax. Only works for single-expression constructors. Makes simple assignments very readable.
ExpressionBodyd.cs
class Point
{
    public double X;
    public double Y;

    // Expression-bodied — only one statement
    public Point(double x, double y) => (X, Y) = (x, y);
}

class Circle
{
    public Point  Center;
    public double Radius;

    // Tuple deconstruction assignment
    public Circle(Point center, double radius)
        => (Center, Radius) = (center, radius);
}

Point  p = new Point(3.0, 4.0);
Circle c = new Circle(p, 5.0);

Constructor Combinations

Multiple constructor patterns can be combined in the same class to cover different creation scenarios. Here are the most useful real-world combinations.

Default + Parameterized Overloading
class Config
{
    public string Host;
    public int    Port;

    // Default
    public Config()
    {
        Host = "localhost";
        Port = 8080;
    }

    // Parameterized
    public Config(string host, int port)
    {
        Host = host;
        Port = port;
    }
}

Config c1 = new Config();
Config c2 = new Config("prod.server", 443);
Parameterized + Chained DRY Pattern
class Logger
{
    public string Name;
    public bool   Verbose;
    public string FilePath;

    // Master — all params
    public Logger(string name, bool verbose, string path)
    { Name=name; Verbose=verbose; FilePath=path; }

    // Chains to master
    public Logger(string name, bool verbose)
        : this(name, verbose, "app.log") { }

    // Chains to master
    public Logger(string name)
        : this(name, false) { }
}
Static + Instance Init + Instance
class AppSettings
{
    public static string Env;
    public string        Section;

    // Static — runs once before first use
    static AppSettings()
    {
        Env = Environment.GetEnvironmentVariable(
                "APP_ENV") ?? "development";
    }

    // Instance — runs per object
    public AppSettings(string section)
    {
        Section = section;
    }
}
Base + Chained Inheritance Chain
class Entity
{
    public int Id;
    public Entity(int id) { Id = id; }
}

class User : Entity
{
    public string Name;
    public string Email;

    // Master: calls base + sets all
    public User(int id, string name, string email)
        : base(id) { Name=name; Email=email; }

    // Chains to master with default email
    public User(int id, string name)
        : this(id, name, "") { }
}
Private + Static Factory Singleton
class CacheManager
{
    private static CacheManager _inst;
    private readonly Dictionary<string,object> _store;

    // Private — prevents external new
    private CacheManager()
    {
        _store = new Dictionary<string,object>();
    }

    // Static factory — single entry point
    public static CacheManager Instance()
        => _inst ??= new CacheManager();
}
Copy + Parameterized Clone Pattern
class Document
{
    public string       Title;
    public List<string> Tags;

    // Parameterized
    public Document(string title)
    {
        Title = title;
        Tags  = new List<string>();
    }

    // Copy — deep clones the list!
    public Document(Document src)
    {
        Title = src.Title;
        Tags  = new List<string>(src.Tags);
    }
}

Full Example

Every keyword and constructor type combined into one cohesive real-world class hierarchy.

FullExample.cs
// ── BASE ABSTRACT CLASS ───────────────────────────────────────
public abstract partial class Animal
{
    public  const    string Kingdom  = "Animalia";    // const
    public  static   int    Count    = 0;             // static — shared
    public  readonly Guid   Id;                       // readonly
    public           string Name     { get; set; }   // auto property
    private          int    _age;                     // private field
    protected        string Species;                  // protected field

    // Static constructor — runs once on first use
    static Animal()
        => Console.WriteLine("[Static] Animal class loaded");

    // Master parameterized constructor
    public Animal(string name, string species, int age)
    {
        Id      = Guid.NewGuid();
        Name    = name;
        Species = species;
        _age    = age;
        Count++;
    }

    // Chained constructor — default age 1
    public Animal(string name, string species)
        : this(name, species, 1) { }

    // Copy constructor
    public Animal(Animal source)
        : this(source.Name, source.Species, source._age) { }

    // Abstract — must override
    public abstract void  Speak();
    // Virtual — can override
    public virtual  void  Move()
        => Console.WriteLine($"{Name} moves");
    // Static utility
    public static   void  ShowCount()
        => Console.WriteLine($"Total animals: {Count}");
}

// ── PARTIAL CLASS (split file) ────────────────────────────────
public abstract partial class Animal
{
    public string GetSummary()
        => $"{Name} ({Species})";
}

// ── SEALED DERIVED CLASS ──────────────────────────────────────
public sealed class Dog : Animal
{
    internal string  Breed { get; }  // internal
    private  bool    _trained;

    // Calls base master ctor + own logic
    public Dog(string name, string breed, int age, bool trained)
        : base(name, "Canis lupus familiaris", age)
    {
        Breed    = breed;
        _trained = trained;
    }

    // Chains to master dog ctor
    public Dog(string name, string breed)
        : this(name, breed, 1, false) { }

    public override void Speak()
        => Console.WriteLine($"{Name}: Woof!");        // override abstract
    public override void Move()
        => Console.WriteLine($"{Name} runs with tail wagging"); // override virtual
}

// ── GENERIC CLASS ─────────────────────────────────────────────
public class AnimalShelter<T> where T : Animal
{
    private readonly List<T> _animals = new();

    public AnimalShelter() { }  // default

    public void  Add(T animal)   => _animals.Add(animal);
    public int   Count()        => _animals.Count;
}

// ── USAGE ────────────────────────────────────────────────────
Dog rex   = new Dog("Rex", "Husky", 3, true);
Dog buddy = new Dog("Buddy", "Labrador");
Dog clone = new Dog("Rex2", rex.Breed);

rex.Speak();           // Rex: Woof!
Animal.ShowCount();    // Total animals: 3

AnimalShelter<Dog> shelter = new();
shelter.Add(rex);
shelter.Add(buddy);
